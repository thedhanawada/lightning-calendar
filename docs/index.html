Quick Start

import { Calendar, Event } from '@forcecalendar/core';

// Create a calendar
const calendar = new Calendar({
  timeZone: 'America/New_York'
});

// Add an event
calendar.addEvent({
  id: 'meeting-1',
  title: 'Team Standup',
  start: new Date('2025-01-15T10:00:00'),
  end: new Date('2025-01-15T10:30:00'),
  category: 'meeting'
});

// Query events
const events = calendar.getEventsForDay(new Date());
